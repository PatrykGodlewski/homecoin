\section{Wprowadzenie}

\begin{frame}{Projekt: HomeCoin}
  \textbf{Cel:} aplikacja webowa do finansów domowych --- dzielenie wydatków, budżety, skarbonki, rozliczenia.

  \vspace{0.6em}
  \begin{columns}[T]
    \column{0.48\textwidth}
    \textbf{Stos technologiczny}
    \begin{itemize}
      \item Go 1.25, PostgreSQL
      \item Superkit + Templ + HTMX (interfejs)
      \item REST API \texttt{/api/v1}
      \item Azure Container Apps (chmura)
    \end{itemize}

    \column{0.48\textwidth}
    \textbf{Odnośniki}
    \begin{itemize}
      \item \href{https://github.com/PatrykGodlewski/homecoin}{GitHub --- kod źródłowy}
      \item \href{https://github.com/PatrykGodlewski/homecoin/blob/main/docs/OCENA.md}{Mapowanie kryteriów oceny}
      \item \href{https://homecoin-api.ashyocean-93c64f80.uaenorth.azurecontainerapps.io/health}{Azure --- działające wdrożenie}
    \end{itemize}
  \end{columns}

  \vspace{0.5em}
  \small\textit{Kryteria: dst $\rightarrow$ dst+ $\rightarrow$ db $\rightarrow$ db+ $\rightarrow$ bdb.}
\end{frame}

\section{Ocena dst}

\begin{frame}{dst --- CRUD + chmura + CI/CD}
  \small
  \begin{tabular}{@{}p{0.28\textwidth}p{0.66\textwidth}@{}}
    \toprule
    \textbf{Kryterium} & \textbf{Implementacja} \\
    \midrule
    Operacje CRUD &
    REST API: wydatki, budżety, kategorie, skarbonki, rozliczenia \\
    & \texttt{GET|POST /api/v1/households/\{id\}/...} \\
    \addlinespace
    Wdrożenie w chmurze &
    Azure: Container Apps + PostgreSQL Flexible Server \\
    & IaC: Terraform (\texttt{infra/terraform/}) + weryfikacja Ansible (\texttt{infra/ansible/}) \\
    \addlinespace
    Potok CI/CD &
    GitHub Actions: \texttt{ci.yml}, \texttt{cd-azure.yml}, \texttt{azure-infra.yml} \\
    \bottomrule
  \end{tabular}

  \vspace{0.5em}
  \normalsize
  \textbf{Demonstracja:} push na \texttt{main} $\rightarrow$ CI $\rightarrow$ CD (rejestr ACR $\rightarrow$ Azure).
\end{frame}

\section{Ocena dst+}

\begin{frame}{dst+ --- bezpieczeństwo}
  \small
  \begin{tabular}{@{}p{0.32\textwidth}p{0.62\textwidth}@{}}
    \toprule
    \textbf{Aspekt} & \textbf{Implementacja} \\
    \midrule
    HTTPS &
    Nginx + TLS lokalnie (\texttt{deploy/docker/nginx/}); na Azure ingress wymusza HTTPS \\
    \addlinespace
    Hasła &
    bcrypt --- \texttt{HashPassword}, \texttt{CheckPassword} \\
    \addlinespace
    Tokeny &
    JWT HMAC-SHA256; refresh tokeny hashowane SHA-256 w DB \\
    \addlinespace
    Nagłówki HTTP &
    HSTS, X-Frame-Options, X-Content-Type-Options \\
    \addlinespace
    Baza w chmurze &
    PostgreSQL Azure: \texttt{sslmode=require} \\
    \addlinespace
    Usługi wewnętrzne &
    Token \texttt{X-Worker-Token} (komunikacja API $\rightarrow$ worker) \\
    \bottomrule
  \end{tabular}
\end{frame}

\begin{frame}[fragile]{dst+ --- demonstracja}
  \textbf{Lokalnie (HTTPS):}
  \begin{lstlisting}
./deploy/docker/nginx/generate-certs.sh
docker compose up --build
curl -k https://127.0.0.1:8081/health
  \end{lstlisting}

  \textbf{Na Azure:}
  \begin{lstlisting}
curl https://homecoin-api.ashyocean-93c64f80.uaenorth.azurecontainerapps.io/health
  \end{lstlisting}

  Odpowiedź: \verb|{"status":"ok"}|

  \vspace{0.4em}
  Hasło użytkownika przechowywane jako hash bcrypt.
\end{frame}

\section{Ocena db}

\begin{frame}{db --- Docker Compose}
  \textbf{Plik:} \texttt{docker-compose.yml} --- 5 kontenerów (postgres, migrate, api, worker, nginx), healthchecki, \texttt{depends\_on}.

  \vspace{0.5em}
  \centering
  \small
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Kontener} & \textbf{Rola} & \textbf{Port (host)} \\
    \midrule
    \texttt{postgres} & Baza danych & 5433 \\
    \texttt{migrate} & Migracje schematu (jednorazowo) & --- \\
    \texttt{api} & REST API + interfejs & wewn.\ 8080 \\
    \texttt{worker} & Zadania w tle & wewn.\ 8080 \\
    \texttt{nginx} & TLS, reverse proxy & 8081 (HTTPS) \\
    \bottomrule
  \end{tabular}

  \vspace{0.5em}
  \normalsize
  \texttt{docker compose up --build} \quad UI: \texttt{https://127.0.0.1:8081}
\end{frame}

\section{Ocena bdb}

\begin{frame}{bdb --- testy w CI/CD}
  \small
  \begin{tabular}{@{}lll@{}}
    \toprule
    \textbf{Typ} & \textbf{Lokalizacja} & \textbf{Zadanie w CI} \\
    \midrule
    Jednostkowe (domena) & \texttt{internal/domain/service/} & Testy jednostkowe \\
    Jednostkowe (auth) & \texttt{internal/infrastructure/auth/} & Testy jednostkowe \\
    Jednostkowe (worker) & \texttt{internal/infrastructure/workerclient/} & Testy jednostkowe \\
    E2E smoke (bash) & \texttt{scripts/ci/smoke\_test.sh} & \textbf{Testy E2E} \\
    E2E (Go) & \texttt{test/e2e/api\_test.go} & \textbf{Testy E2E} \\
    \bottomrule
  \end{tabular}

  \vspace{0.5em}
  \normalsize
  \textbf{Potok E2E:} certyfikaty TLS $\rightarrow$ \texttt{docker compose up --wait} $\rightarrow$ smoke test $\rightarrow$ \texttt{go test -tags=e2e}
\end{frame}
